// 一軍登録: 最大29人 / 抹消後10日間は再登録不可
// 外国人: 一軍は合計3人（投手最大2、野手最大2）まで
function canPromote(t,p){
const ichi=t.ichiCount;
if(ichi>=29)return{ok:false,reason:‘一軍登録は最大29人です（現在’+ichi+‘人）’};
if(p.isIku)return{ok:false,reason:‘育成選手は支配下昇格が必要です’};
if(p.reg===‘ichi’)return{ok:false,reason:‘すでに一軍登録中です’};
if(p.demotedDay!==null&&G.day-p.demotedDay<10)return{ok:false,reason:‘抹消後10日間は再登録不可（残’+(10-(G.day-p.demotedDay))+‘日）’};
if(p.isForeign){
const fi=t.foreignIchiCount,fp=t.foreignIchiP,fh=t.foreignIchiH;
// NPB外国人枠（+1適用後）: 合計5人まで、投手最大3人・野手最大3人
if(fi>=5)return{ok:false,reason:‘外国人の一軍登録は最大5人です’};
if(p.isPitcher&&fp>=3)return{ok:false,reason:‘外国人投手は最大3人までです’};
if(!p.isPitcher&&fh>=3)return{ok:false,reason:‘外国人野手は最大3人までです’};
}
return{ok:true,reason:’’};
}

// ════ FINANCE ════
function calcFin(t){
const wp=t.wins/Math.max(1,t.wins+t.losses),sc=t.fans/100;
const avg=Math.round(t.cap*(0.6+wp*0.35)*sc);
const ticketRev=avg*71*ri(2800,3500);
const tvRev=Math.round((5+sc*45)*OKU*(0.8+wp*0.4));
const goodsRev=Math.round((2+sc*18)*OKU*(0.7+wp*0.3));
const sponsorRev=(t.sponsors||[]).reduce((s,sp)=>s+sp.fee,0);
const otherRev=Math.round(ri(1,5)*OKU);
const totalRev=ticketRev+tvRev+goodsRev+sponsorRev+otherRev;
const payroll=t.payroll;
const staffCost=(t.staff||{}).staffCost||ri(5,20)*OKU;
const facility=Math.round(ri(4,14)*OKU);
const travel=Math.round(ri(3,5)*OKU);
const dep=t.depreciation||ri(1,5)*OKU;
const mktg=Math.round(ri(1,4)*OKU*sc);
const totalExp=payroll+staffCost+facility+travel+dep+mktg;
return{ticketRev,tvRev,goodsRev,sponsorRev,otherRev,totalRev,payroll,staffCost,facility,travel,dep,mktg,totalExp,profit:totalRev-totalExp,avg,wp};
}

function runLottery(){

const sorted=[…G.teams].sort((a,b)=>b.wins-a.wins),lT=sorted.slice(-7),w=[140,105,77,57,40,27,18],lOrd=[],rem=[…lT];
while(rem.length>0){let tot=0;for(let j=0;j<rem.length;j++)tot+=(w[j]||10);let r2=Math.random()*tot,cum=0,picked=false;for(let j=0;j<rem.length;j++){cum+=(w[j]||10);if(r2<=cum){lOrd.push(rem.splice(j,1)[0].idx);picked=true;break;}}if(!picked)lOrd.push(rem.splice(0,1)[0].idx);}
const rest=sorted.slice(0,-7).reverse().map(t=>t.idx);
G.draftLottery={order:[…lOrd,…rest],currentPick:0,round:1,maxRound:3,done:false};
const myN=G.draftLottery.order.indexOf(G.myIdx)+1;
addNews(‘ドラフトロッタリー結果’,’1位指名権: ‘+G.teams[G.draftLottery.order[0]].name+’\nあなたの指名順位: ’+myN+‘位’,‘major’);

function cpuDraftTick(){

if(G.day<289)return; // ドラフト会議前は進行しない
const dl=G.draftLottery,ord=dl.order,total=ord.length*dl.maxRound;
while(dl.currentPick<total){
const ri2=Math.floor(dl.currentPick/ord.length),pi=dl.currentPick%ord.length;
const tIdx=(ri2%2===0)?ord[pi]:ord[ord.length-1-pi];
if(tIdx===G.myIdx)return;
const t=G.teams[tIdx],av=G.draftProspects.filter(p=>!p.drafted);
if(av.length&&t.players.length<70){av.sort((a,b)=>b.ovr-a.ovr);const pick=av[0];pick.drafted=true;pick.contract=makeCon(pick.ovr,pick.age,false,false);pick.contract.years=1;pick.reg=‘ni’;t.players.push(pick);addLog(t.abbr+‘→’+pick.name+‘指名’,‘info’);}
dl.currentPick++;
}
dl.done=true;G.draftProspects.filter(p=>!p.drafted).forEach(p=>{p.contract=makeCon(p.ovr,p.age,false,false);G.freeAgents.push(p);});
addNews(‘ドラフト終了’,‘全指名が完了。指名漏れ選手はFA市場へ。’,‘event’);
}

function simGames(days){
if(G.day<62||G.day>251)return;
const gpd=143/(251-62),n=Math.max(1,Math.round(gpd*days));
const myT=G.teams[G.myIdx];
for(let g=0;g<n;g++){
for(let i=0;i<G.teams.length;i+=2){
const t1=G.teams[i],t2=G.teams[Math.min(i+1,G.teams.length-1)];
if(t1===t2)continue;
const p2=clamp(.5+(teamOvr(t1)-teamOvr(t2))*.008,.2,.8);
const rng=Math.random();
let result;
if(rng<p2*.95){t1.wins++;t2.losses++;result=t1===myT?‘win’:t2===myT?‘loss’:null;}
else if(rng<p2*.95+.05){t1.draws++;t2.draws++;result=(t1===myT||t2===myT)?‘draw’:null;}
else{t1.losses++;t2.wins++;result=t1===myT?‘loss’:t2===myT?‘win’:null;}
// 自チームの試合は詳細記録
if(result&&(t1===myT||t2===myT)){
const opp=t1===myT?t2:t1;
const myS=ri(0,9),opS=result===‘win’?ri(0,myS-1):result===‘loss’?ri(myS+1,9):myS;
const month=G.day<93?3:G.day<124?4:G.day<154?5:G.day<185?6:G.day<215?7:G.day<246?8:9;
const key=G.season+’_’+month;
if(!G.monthlyStats)G.monthlyStats={};
if(!G.monthlyStats[key])G.monthlyStats[key]={wins:0,losses:0,draws:0,totalAtt:0,games:0,month,season:G.season};
const ms=G.monthlyStats[key];
const att=Math.round(myT.cap*(0.55+Math.random()*0.4)*(myT.fans/100));
ms.games++;ms.totalAtt+=att;
if(result===‘win’)ms.wins++;else if(result===‘loss’)ms.losses++;else ms.draws++;
if(!G.gameLog)G.gameLog=[];
G.gameLog.push({day:G.day,season:G.season,month,opp:opp.abbr,result,att,myS:Math.min(myS,15),opS:Math.min(opS,15)});
if(G.gameLog.length>500)G.gameLog.shift();
}
}
}
// 一軍選手のスタッツ更新
myT.players.filter(p=>p.reg===‘ichi’).forEach(p=>{if(p.isPitcher)simPI(p,n/5);else simHI(p,n);});

simFarmGames(days);
// 週次ドラマイベント
if(n>=7&&Math.random()<0.6)genWeeklyDrama();
}
function simHI(p,games){
const a=p.abilities;
const pa=Math.round(4.1*games);
// NPBリアル打率: .210-.350
// NPB2024平均打率.241, 首位打者.330前後
const avg=clamp(.215+a.meet/100*.13+rand(-.008,.008),.210,.345);
const bbR=clamp(.055+a.eye/100*.07,.040,.135);
const kR=clamp(.155-a.meet/100*.08,.065,.265);
const ab=Math.round(pa*(1-bbR-.012));
const h=Math.round(ab*avg);
// 本塁打: NPB大砲30本、平均5-10本
// NPB2024: チーム90-130本。主軸20-30本、平均7-12本
const hrRate=clamp(a.power/100*.018+rand(-.002,.002),.002,.024);
const hr=Math.round(ab*hrRate);
// 二塁打: 安打の15-25%
const d2=Math.round(h*clamp(.16+a.power/100*.08,.12,.27));
// 三塁打: 安打の1-3%
const d3=Math.round(h*clamp(a.spd/100*.025,.003,.030));
const sing=Math.max(0,h-hr-d2-d3);
// 打点: HR+進塁打ベース
// NPB主軸は80-100打点、平均35-60点
const rbi=Math.round(hr*1.90+sing*.22+d2*.32+d3*.40+rand(-3,3));
// 盗塁: NPB平均5-10、俊足20-35
const sb=Math.round(clamp(a.spd/100*28-8+rand(-2,2),0,35)*(games/143));
// 犠打
const sac=Math.round(clamp((100-a.power)/100*.15+(100-a.meet)/100*.08,.0,.25)*games*.18);
// 得点
const runs=Math.round((clamp(.230+a.eye/100*.12,.290,.420)*.8+a.spd/100*.15)*pa*.14);
const s=p.stats;
s.g+=Math.round(games);s.pa+=pa;s.ab+=ab;s.h+=h;s.hr+=hr;s.rbi+=rbi;
s.bb+=Math.round(pa*bbR);s.k+=Math.round(pa*kR);s.sb+=sb;s.sbA+=Math.round(sb*1.25);
s.d2=(s.d2||0)+d2;s.d3=(s.d3||0)+d3;s.sac=(s.sac||0)+sac;s.runs=(s.runs||0)+runs;
if(s.ab>0){s.avg=s.h/s.ab;const slg=(sing+d2*2+d3*3+hr*4)/s.ab;s.slg=slg;}
const obp=clamp((s.h+(s.bb||0))/Math.max(1,s.pa),.290,.420);
s.obp=obp;if(s.pa>0){s.kPct=s.k/s.pa;s.bbPct=s.bb/s.pa;}
s.ops=clamp(obp+(s.slg||.350),.500,.980);
s.xAvg=clamp(s.avg+rand(-.012,.012),.195,.360);
s.xSlg=clamp((s.slg||.350)+rand(-.025,.025),.290,.620);
s.hardHit=clamp(a.power/100*.52+rand(-.04,.04),.15,.60);
s.barrel=clamp(a.power/100*a.meet/100*.22+rand(-.015,.015),.005,.22);
s.sprintSpd=clamp(24+a.spd/100*10+rand(-.4,.4),23,36);
s.rangeScore=clamp(a.spd/100*8+a.def/100*12+rand(-.8,.8),0,20);
s.throwAcc=clamp(a.arm/100*100+rand(-4,4),10,99);
}
function simPI(p,games){
const a=p.abilities;
const isSP=p.pos===‘SP’,isCL=p.pos===‘CL’;
let gamesApp,ipPerApp;
if(isSP){
gamesApp=Math.round((games/143)*clamp(23+a.stm/100*3,20,28));
ipPerApp=clamp(4.5+a.stm/100*2.0+a.ctrl/100*.6,3.0,7.5); // 平均5.5-7回
}else if(isCL){
gamesApp=Math.round((games/143)*clamp(45+a.ctrl/100*10,35,55));
ipPerApp=1.0;
}else{
const isWin=p.pitchRole===‘win’||p.pitchRole===‘win2’;
gamesApp=Math.round((games/143)*(isWin?clamp(50+a.ctrl/100*8,40,60):clamp(35+ri(0,15),25,50)));
ipPerApp=clamp(.8+a.stm/100*.8,.5,2.0);
}
const ip=clamp(gamesApp*ipPerApp*rand(.85,1.05),0,isSP?178:isCL?56:78); // NPBエース160-175回
const era=clamp(8.5-a.ctrl/100*4.5-a.brk/100*1.5-a.spd/100*.8+rand(-.3,.4),1.20,8.00);
const k9=clamp(3.0+a.brk/100*5.0+a.spd/100*1.2,2.8,9.8); // NPBリアル: 平均5.5-8, 最高9-10
const bb9=clamp(4.5-a.ctrl/100*3.2,.5,5.5);
const h9=clamp(9.5-a.brk/100*3.5-a.ctrl/100*1.5,5.0,12.5);
const s=p.stats;
s.g+=gamesApp;if(isSP)s.gs+=gamesApp;s.ip+=ip;
s.k+=Math.round(k9*ip/9);s.bb+=Math.round(bb9*ip/9);
s.h+=Math.round(h9*ip/9);s.er+=Math.round(era*ip/9);
s.hr+=Math.round(ip/9*clamp(a.brk/100*.8,.3,1.8));
if(isSP){const wp=clamp(.5+(5-era)/10,.25,.75);s.w+=Math.round(gamesApp*wp*.68);s.l+=Math.round(gamesApp*(1-wp)*.68);}
if(isCL)s.sv+=Math.round(gamesApp*.78);
if(s.ip>0){s.era=s.er/s.ip*9;s.whip=(s.bb+s.h)/s.ip;s.k9=s.k/s.ip*9;s.bb9=s.bb/s.ip*9;}
s.xera=clamp(s.era+rand(-.35,.35),1.0,9.0);
s.gbPct=clamp((100-a.spd)/100*.48+rand(-.04,.04),.28,.68);
s.whiffPct=clamp(a.brk/100*.42+rand(-.03,.03),.08,.52);
s.velo=p.kmh;s.spinAvg=ri(1800,2650);
s.hardHitA=clamp((100-a.ctrl)/100*.48+rand(-.04,.04),.10,.65);
}

// 投手適性スコア計算
function pitcherFit(p,role){
const a=p.abilities;
if(role.startsWith(‘sp’)){
// 先発適性: スタミナ・制球が重要、変化球もある程度
return Math.round(a.stm*0.35+a.ctrl*0.3+a.brk*0.2+a.spd*0.15);
}else if(role===‘cl’){
// 抑え: 制球・球速・対右対左が重要、スタミナ不要
return Math.round(a.ctrl*0.3+a.spd*0.3+a.brk*0.25+(a.vsL+a.vsR)/2*0.15);
}else if(role===‘win’||role===‘win2’){
return Math.round(a.ctrl*0.3+a.spd*0.25+a.brk*0.25+a.vsR*0.2);
}else if(role===‘left’){
return Math.round(a.vsL*0.4+a.ctrl*0.3+a.brk*0.2+a.spd*0.1);
}else{
return Math.round(a.ctrl*0.25+a.stm*0.3+a.brk*0.2+a.spd*0.25);
}
}

function initDragSort(container,t){
let dragEl=null,startY=0,origIdx=0,clone=null;
const rows=()=>[…container.children];

container.addEventListener(‘touchstart’,e=>{
const row=e.target.closest(’[data-pid]’);if(!row)return;
dragEl=row;startY=e.touches[0].clientY;origIdx=rows().indexOf(row);
// ドラッグ中のスタイル
row.style.opacity=‘0.5’;row.style.background=‘rgba(204,20,37,.08)’;
// クローンで位置表示
clone=row.cloneNode(true);clone.style.cssText=‘position:fixed;left:’+row.getBoundingClientRect().left+‘px;top:’+row.getBoundingClientRect().top+‘px;width:’+row.offsetWidth+‘px;z-index:999;opacity:.9;pointer-events:none;background:#181c27;border-radius:8px;box-shadow:0 4px 16px rgba(0,0,0,.6);’;
document.body.appendChild(clone);
},{passive:true});

container.addEventListener(‘touchmove’,e=>{
if(!dragEl||!clone)return;
e.preventDefault();
const y=e.touches[0].clientY;
clone.style.top=(y-30)+‘px’;
// 挿入位置を計算
const allRows=rows();
for(let i=0;i<allRows.length;i++){
const r=allRows[i];if(r===dragEl)continue;
const rc=r.getBoundingClientRect();
if(y<rc.top+rc.height/2){container.insertBefore(dragEl,r);break;}
if(i===allRows.length-1)container.appendChild(dragEl);
}
},{passive:false});

container.addEventListener(‘touchend’,()=>{
if(!dragEl)return;
if(clone){clone.remove();clone=null;}
dragEl.style.opacity=’’;dragEl.style.background=’’;
// 新しい順番で打順を再設定
rows().forEach((r,i)=>{
const p=t.players.find(x=>x.id===r.dataset.pid);
if(p){p.batOrder=i+1;}
});
// 番号表示を更新
rows().forEach((r,i)=>{
const numSpan=r.querySelectorAll(‘span’)[1];
if(numSpan)numSpan.textContent=(i+1)+‘番’;
});
dragEl=null;
autoSave();
},{passive:true});
}

function submitTradeProposal(el){
const offer=G.tradeOffer;
if(!offer||offer.targetTeamIdx<0){toast(‘交渉相手を選んでください’,‘bad’);return;}
const myT=G.teams[G.myIdx],tt=G.teams[offer.targetTeamIdx];
const myP=offer.myPids.map(id=>myT.players.find(x=>x.id===id)).filter(Boolean);
const theirP=offer.targetPids.map(id=>tt.players.find(x=>x.id===id)).filter(Boolean);
if(theirP.length===0){toast(‘相手から欲しい選手を選んでください’,‘bad’);return;}

// 承諾判定（OVRバランス+乱数+チーム状況）
const myVal=myP.reduce((s,p)=>s+p.ovr,0)+(offer.myPick?pickValueOvr(offer.myPick):0);
const theirVal=theirP.reduce((s,p)=>s+p.ovr,0);
const ratio=myVal/Math.max(theirVal,1);
// 相手がAクラスなら駆け引きに応じる確率が低い
const ttDiv=G.teams.filter(x=>x.div===tt.div).sort((a,b)=>b.wins-a.wins);
const ttRank=ttDiv.findIndex(x=>x.idx===tt.idx)+1;
const isContender=ttRank<=2;
// 成立確率: 均衡(ratio~1.0)→70%, 相手有利(ratio>=1.1)→85%, 自分有利(ratio<=0.8)→30%
const baseProb=ratio>=1.1?0.85:ratio>=0.95?0.70:ratio>=0.85?0.55:0.30;
const finalProb=clamp(baseProb+(isContender?-0.10:0.05)+rand(-0.10,0.10),0.05,0.92);

if(Math.random()<finalProb){
// 成立
myP.forEach(p=>{myT.players=myT.players.filter(x=>x.id!==p.id);p.reg=‘ni’;tt.players.push(p);});
theirP.forEach(p=>{tt.players=tt.players.filter(x=>x.id!==p.id);p.reg=‘ni’;myT.players.push(p);});
const desc=’【’+myT.abbr+‘→’+tt.abbr+’】’+myP.map(p=>p.name).join(’+’)+
(offer.myPick?’+’+offer.myPick:’’)+’\n【’+tt.abbr+‘→’+myT.abbr+’】’+theirP.map(p=>p.name).join(’+’);
addNews(‘トレード成立’,desc,‘good’);
toast(‘トレード成立！’,‘good’);
G.tradeOffer={myPids:[],myPick:null,targetTeamIdx:-1,targetPids:[]};
}else{
const reason=ratio<0.85?‘提示価値が低すぎます’:‘相手が現在の戦力を手放したくない状況のようです’;
addNews(‘トレード拒否’,tt.name+‘がトレードを拒否しました。’+reason,‘warn’);
toast(‘トレード拒否（成功率’+Math.round(finalProb*100)+’%）’,‘bad’);
}
renderAll();
}

function executeTrade(myPid,theirPid,theirTeamIdx){

const myP=myT.players.find(x=>x.id===myPid),theirP=theirT.players.find(x=>x.id===theirPid);
if(!myP||!theirP){toast(‘選手情報エラー’,‘bad’);return;}
// CPUの承諾判定: OVR差・年俸バランスを見る
const ovrDiff=Math.abs(myP.ovr-theirP.ovr);const salDiff=Math.abs(myP.contract.salary-theirP.contract.salary);
const acceptProb=ovrDiff<=5?0.65:ovrDiff<=10?0.45:0.25;
if(Math.random()<acceptProb){
// トレード成立
myT.players=myT.players.filter(x=>x.id!==myPid);
theirT.players=theirT.players.filter(x=>x.id!==theirPid);
theirP.reg=‘ni’;myT.players.push(theirP);
myP.reg=‘ni’;theirT.players.push(myP);
addNews(‘トレード成立’,myT.abbr+‘→’+theirT.abbr+’: ‘+myP.name+’\n’+theirT.abbr+‘→’+myT.abbr+’: ‘+theirP.name,‘good’);

}else{
addNews(‘トレード拒否’,theirT.name+‘が’+myP.name+‘とのトレードを拒否しました。’,‘warn’);
toast(theirT.abbr+‘がトレードを拒否しました’,‘bad’);
}
renderAll();
}
